#!/bin/bash

# Sourcing this to gain access to text centering helpers
source /usr/lib/shark_text_helpers

# Alert usage:
# BLOCKING_ALERT "Title" (optional timeout) (optional flash flag) (optional raw input flag)
# optional arguments 2 and 3 require the inclusion of previous arguments, see examples below

# NOTE: similar to ALERT, BLOCKING_ALERT _can_ accept -1 as a timeout input. THIS IS NOT RECOMMENDED;
# using no timeout will require your payload to background it, and manage its lifecycle by killing its PID.

# Put something in the payload log
SCREEN_WRITE "Blocking Alert Demo!"

########################################################
# Flashing Alert
################################################################################

SCREEN_WRITE "Flashing"

sleep 1 # give SCREEN_WRITE a chance to render before drawing alert over it

# This will display a flashing alert titled "Flashing!" for 2 seconds, USER CANNOT EXIT EARLY
# the title will automatically be centered horizontally and vertically
BLOCKING_ALERT "Flashing!" 3

SCREEN_WRITE "Done"

sleep 1

########################################################
# Solid Alert
################################################################################

SCREEN_WRITE "Solid"

sleep 1 # give SCREEN_WRITE a chance to render before drawing alert over it

# This will display a solid alert titled "Solid" for 1 second, USER CANNOT EXIT EARLY
# the title will automatically be centered horizontally and vertically
BLOCKING_ALERT "Solid" 3 false

SCREEN_WRITE "Done"

sleep 1

########################################################
# Raw Formatting Alert (Raw)
################################################################################

SCREEN_WRITE "Raw Title"

sleep 1 # give SCREEN_WRITE a chance to render before drawing alert over it

# This will display a solid alert titled "Raw Format!" for 1 second, USER CANNOT EXIT EARLY
# the title will not be automatically formatted by the BLOCKING_ALERT command, allowing the caller to format as desired
BLOCKING_ALERT "Raw Format!" 3 false true

SCREEN_WRITE "Done"

sleep 1

########################################################
# Raw Formatting Alert (Formatted)
################################################################################

SCREEN_WRITE "Formatting"

sleep 1 # give SCREEN_WRITE a chance to render before drawing alert over it

# This will display a solid alert titled "Raw Format!" for 1 second, USER CANNOT EXIT EARLY
BLOCKING_ALERT "$(center_text "Manual")\n\n$(center_text "Format!")" 3 false true

SCREEN_WRITE "Done"

sleep 1

SCREEN_WRITE "Demo Complete"
