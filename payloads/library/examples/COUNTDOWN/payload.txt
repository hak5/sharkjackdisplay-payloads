#!/bin/bash

# Display a cancellable countdown to the user.
# Press button to cancel, or let it timeout

resp="$(COUNTDOWN "COUNTDOWN" 10)"
if [ "$resp" = "Cancelled" ]; then
  ALERT "$resp" 2 false
  SCREEN_WRITE "Cancelled!"
else
  # $resp will be "Timeout"
  ALERT "$resp" 2 false
  SCREEN_WRITE "Timeout!"
fi
